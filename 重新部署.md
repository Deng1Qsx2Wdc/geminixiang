# 第一步：彻底清理服务器
#!/bin/bash
# 彻底清理 Gemini API 项目

echo "=========================================="
echo "  清理所有 Gemini API 项目"
echo "=========================================="
echo ""

# 1. 停止所有相关服务
echo "[1/6] 停止所有服务..."
sudo systemctl stop gemini-api 2>/dev/null
sudo systemctl disable gemini-api 2>/dev/null
sudo systemctl daemon-reload

# 2. 停止所有相关进程
echo "[2/6] 停止相关进程..."
pkill -f "server.py" 2>/dev/null
sleep 2

# 3. 删除 systemd 服务文件
echo "[3/6] 删除服务配置..."
sudo rm -f /etc/systemd/system/gemini-api.service
sudo systemctl daemon-reload

# 4. 查找并删除所有相关目录
echo "[4/6] 查找项目目录..."
echo "  查找 server.py 文件..."
find /root -name "server.py" -type f 2>/dev/null | grep -v venv | grep -v site-packages
find /opt -name "server.py" -type f 2>/dev/null | grep -v venv | grep -v site-packages
find /home -name "server.py" -type f 2>/dev/null | grep -v venv | grep -v site-packages

# 5. 删除所有相关目录
echo "[5/6] 删除项目目录..."
sudo rm -rf /opt/gemini-api
rm -rf /root/geminixiang
rm -rf /root/gemininixiang
find /root -type d -name "*gemini*" -o -name "*gemininixiang*" 2>/dev/null | xargs rm -rf 2>/dev/null
find /opt -type d -name "*gemini*" 2>/dev/null | xargs sudo rm -rf 2>/dev/null

# 6. 验证清理结果
echo "[6/6] 验证清理结果..."
echo "  检查进程："
ps aux | grep -E "server.py|gemini" | grep -v grep || echo "  ✓ 无相关进程"
echo "  检查服务："
sudo systemctl list-units --type=service | grep -i gemini || echo "  ✓ 无相关服务"
echo "  检查端口："
sudo lsof -i :8001 2>/dev/null || echo "  ✓ 端口 8001 未被占用"
echo "  检查目录："
ls -d /opt/gemini-api /root/geminixiang /root/gemininixiang 2>/dev/null || echo "  ✓ 目录已删除"

echo ""
echo "=========================================="
echo "  清理完成！"
echo "=========================================="

# 第二步：从 zip 文件重新部署项目

#!/bin/bash
# 从 zip 文件部署 Gemini API 项目

echo "=========================================="
echo "  部署 Gemini API 项目（从 zip 文件）"
echo "=========================================="
echo ""

# 1. 创建项目目录
echo "[1/7] 创建项目目录..."
mkdir -p /opt/gemini-api
cd /opt/gemini-api

# 2. 下载或使用 zip 文件
echo "[2/7] 获取项目文件..."
ZIP_FILE="/tmp/gemininixiang.zip"

# 检查是否已有 zip 文件
if [ -f "/root/geminixiang/gemininixiang.zip" ]; then
    echo "  使用服务器上的 zip 文件..."
    cp /root/geminixiang/gemininixiang.zip $ZIP_FILE
elif [ -f "$ZIP_FILE" ]; then
    echo "  使用临时目录的 zip 文件..."
else
    echo "  从 GitHub 下载 zip 文件..."
    wget https://github.com/erxiansheng/gemininixiang/archive/refs/heads/main.zip -O $ZIP_FILE
    if [ $? -ne 0 ]; then
        echo "  下载失败，尝试直接下载 zip 文件..."
        # 如果 main.zip 不存在，可能需要直接下载仓库的 zip
        wget https://github.com/erxiansheng/gemininixiang/archive/refs/heads/main.zip -O $ZIP_FILE || {
            echo "  错误：无法下载 zip 文件"
            echo "  请手动上传 gemininixiang.zip 到服务器，然后重新运行此脚本"
            exit 1
        }
    fi
fi

# 3. 解压文件
echo "[3/7] 解压项目文件..."
if [ -f "$ZIP_FILE" ]; then
    unzip -q $ZIP_FILE -d /tmp/
    # 检查解压后的目录结构
    if [ -d "/tmp/gemininixiang-main" ]; then
        # GitHub 下载的格式
        cp -r /tmp/gemininixiang-main/* /opt/gemini-api/
        rm -rf /tmp/gemininixiang-main
    elif [ -d "/tmp/gemininixiang" ]; then
        # 直接解压的格式
        cp -r /tmp/gemininixiang/* /opt/gemini-api/
        rm -rf /tmp/gemininixiang
    else
        # 查找解压后的目录
        EXTRACTED_DIR=$(find /tmp -maxdepth 1 -type d -name "*gemininixiang*" | head -1)
        if [ -n "$EXTRACTED_DIR" ]; then
            cp -r $EXTRACTED_DIR/* /opt/gemini-api/
            rm -rf $EXTRACTED_DIR
        else
            echo "  错误：无法找到解压后的目录"
            exit 1
        fi
    fi
    rm -f $ZIP_FILE
    echo "  ✓ 文件已解压"
else
    echo "  错误：未找到 zip 文件"
    exit 1
fi

# 4. 创建虚拟环境
echo "[4/7] 创建虚拟环境..."
python3 -m venv venv
source venv/bin/activate

# 5. 安装依赖
echo "[5/7] 安装依赖..."
pip install --upgrade pip -q
pip install -r requirements.txt -q
echo "  ✓ 依赖已安装"

# 6. 创建 systemd 服务
echo "[6/7] 创建系统服务..."
sudo tee /etc/systemd/system/gemini-api.service > /dev/null <<EOF
[Unit]
Description=Gemini OpenAI API Server
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/opt/gemini-api
Environment="PATH=/opt/gemini-api/venv/bin"
ExecStart=/opt/gemini-api/venv/bin/python /opt/gemini-api/server.py
Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal

[Install]
WantedBy=multi-user.target
EOF

# 7. 配置防火墙和启动服务
echo "[7/7] 配置防火墙并启动服务..."
sudo ufw allow 8001/tcp 2>/dev/null || sudo firewall-cmd --add-port=8001/tcp --permanent 2>/dev/null || echo "  防火墙配置跳过"
sudo systemctl daemon-reload
sudo systemctl enable gemini-api
sudo systemctl start gemini-api

# 等待并检查状态
sleep 3
echo ""
echo "=========================================="
echo "  部署完成！"
echo "=========================================="
echo ""
sudo systemctl status gemini-api --no-pager

# 获取服务器 IP
SERVER_IP=$(hostname -I | awk '{print $1}')
if [ -z "$SERVER_IP" ]; then
    SERVER_IP="服务器IP"
fi

echo ""
echo "服务信息："
echo "  本地访问: http://localhost:8001/admin"
echo "  外部访问: http://$SERVER_IP:8001/admin"
echo "  API 地址: http://$SERVER_IP:8001/v1"
echo "  API Key:  sk-gemini"
echo ""
echo "下一步："
echo "  1. 访问后台配置页面配置 Cookie"
echo "  2. 如果图片上传失败，配置网络代理（美国服务器通常不需要）"
echo "  3. 测试 API 调用"
echo ""
echo "常用命令："
echo "  查看状态: sudo systemctl status gemini-api"
echo "  查看日志: sudo journalctl -u gemini-api -f"
echo "  重启服务: sudo systemctl restart gemini-api"
